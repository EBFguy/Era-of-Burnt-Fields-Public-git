law_change_effect_list = {
	if = {
		limit = { check_variable = { ebf_mobilization_category = 1 } }
		law_change_conscription_effect_list = YES
	}
	else_if = {
		limit = { check_variable = { ebf_mobilization_category = 2 }  }
		law_change_training_standard_effect_list = YES
	}

law_change_conscription_effect_list = { #GUIDED
	if = { #first check, just in case they select the already selected law 
		limit = { NOT = { check_variable = { EBF_conscription_current_law = mobilization_1_law_ladder } } }
		if = { #second check, checks law is within allowed laws
			limit = {
				check_variable = {
					var = mobilization_1_law_ladder
					value = conscription_allow_law_min
					compare = greater_than_or_equals
				}
				check_variable = {
					var = mobilization_1_law_ladder
					value = conscription_allow_law_max
					compare = less_than_or_equals
				}
			}
			if = {#final checks, checks for PP and gives effects
				limit = {
					check_variable = {mobilization_1_law_ladder = 8}
					has_political_power > EBF_conscription_law_cost_var^8
				}
				add_political_power = EBF_conscription_law_cost^8
				add_ideas = EBF_scraping_the_barrel
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 7}
					has_political_power > EBF_conscription_law_cost_var^7
				}
				add_political_power = EBF_conscription_law_cost^7
				add_ideas = EBF_all_adults_serve
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 6}
					has_political_power > EBF_conscription_law_cost_var^6
				}
				add_political_power = EBF_conscription_law_cost^6
				add_ideas = EBF_service_by_requirement
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 5}
					has_political_power > EBF_conscription_law_cost_var^5
				}
				add_political_power = EBF_conscription_law_cost^5
				add_ideas = EBF_four_year_draft
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 4}
					has_political_power > EBF_conscription_law_cost_var^4
				}
				add_political_power = EBF_conscription_law_cost^4
				add_ideas = EBF_two_year_draft
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 3}
					has_political_power > EBF_conscription_law_cost_var^3
				}
				add_political_power = EBF_conscription_law_cost^3
				add_ideas = EBF_one_year_draft
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 2}
					has_political_power > EBF_conscription_law_cost_var^2
				}
				add_political_power = EBF_conscription_law_cost^2
				add_ideas = EBF_volunteer_only
			}
			else_if = {
				limit = {
					check_variable = {mobilization_1_law_ladder = 1}
					has_political_power > EBF_conscription_law_cost_var^1
				}
				add_political_power = EBF_conscription_law_cost^1

ip_maths = {
	set_temp_variable = { EBF_maxIPs = standard_IPs }
	add_to_temp_variable = { EBF_maxIPs = modifier@added_military_ip }
	add_to_temp_variable = { EBF_maxIPs = added_military_ip }
	subtract_from_temp_variable = { EBF_maxIPs = modifier@remove_military_ip }
	subtract_from_temp_variable = { EBF_maxIPs = remove_military_ip }
	set_temp_variable = { EBF_currentIPs = ground_ips_used }
	add_to_temp_variable = { EBF_currentIPs = naval_ips_used }
	add_to_temp_variable = { EBF_currentIPs = air_ips_used }
	add_to_temp_variable = { EBF_currentIPs = research_ips_used }
	add_to_temp_variable = { EBF_currentIPs = civil_services_ips_used }
	set_temp_variable = {
		remaining_IPs = EBF_maxIPs
	}
	subtract_from_temp_variable = {
		remaining_IPs = EBF_currentIPs
	}
}

update_ground_investment = {
	clear_variable = total_ground_ips_used
	set_variable = { total_ground_ips_used = added_military_ip_ground }
	add_to_variable = { total_ground_ips_used = modifier@added_military_ip_ground }
	set_variable = { all_added_military_ip_ground = total_ground_ips_used }
	add_to_variable = {total_ground_ips_used = ground_ips_used}

	clamp_variable = { var = total_ground_ips_used min = all_added_military_ip_ground max = 100 }
}

change_ground_investment = {
	ip_maths = yes
	if = {
		limit = { check_variable = { ip_change < 0 } }
		add_to_variable = { ground_ips_used = ip_change }
	}
	else = {
		clamp_temp_variable = { var = ip_change min = 0 max = remaining_IPs }
		add_to_variable = { ground_ips_used = ip_change }
	}
	clamp_variable = { var = ground_ips_used min = 0 max = 100 }
}

upgrade_consciption_type = {
	set_temp_variable = { maths2 = EBF_conscription_current_law }
	add_to_temp_variable = { maths2 = 1 }
	if = {
		limit = { check_variable = { maths2 > conscription_allow_law_max } }
		custom_effect_tooltip = current_conscription_type_cannot_go_higher
		add_political_power = 30
	}
	else = {
		if = {
			limit = {
				has_idea = EBF_all_adults_serve
			}
			add_ideas = scraping_the_barrel
		}
		else_if = {
			limit = {
				has_idea = EBF_service_by_requirement
			}
			add_ideas = EBF_all_adults_serve
		}
		else_if = {
			limit = {
				has_idea = EBF_four_year_draft
			}
			add_ideas = EBF_service_by_requirement
		}
		else_if = {
			limit = {
				has_idea = EBF_two_year_draft
			}
			add_ideas = EBF_four_year_draft
		}
		else_if = {
			limit = {
				has_idea = EBF_one_year_draft
			}
			add_ideas = EBF_two_year_draft
		}
		else_if = {
			limit = {
				has_idea = EBF_volunteer_only
			}
			add_ideas = EBF_one_year_draft
		}
		else_if = {
			limit = {
				has_idea = EBF_disarmed_nation
			}
			add_ideas = EBF_volunteer_only
		}		
	}
}
